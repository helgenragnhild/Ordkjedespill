import pygame import random import sys # Initialize Pygame pygame.init() # Skjermstørrelse WIDTH, HEIGHT = 800, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Ordkjede-spillet") # Farger WHITE = (255, 255, 255) BLACK = (0, 0, 0) GREEN = (0, 255, 0) RED = (255, 0, 0) # Font font = pygame.font.SysFont(None, 48) # Ordlist ordliste = [ "skole", "eple", "elev", "vindu", "bok", "katt", "tavle", "elefant", "tromme", "mus", "sol", "lampe" ] def velg_startord(ordliste): return random.choice(ordliste) def draw_text(text, color, x, y): img = font.render(text, True, color) screen.blit(img, (x, y)) def main(): clock = pygame.time.Clock() poeng = 0 brukt_ord = [] current_word = velg_startord(ordliste) brukt_ord.append(current_word) input_text = "" message = f"Startord: {current_word}" running = True while running: screen.fill(WHITE) for event in pygame.event.get(): if event.type == pygame.QUIT: running = False break if event.type == pygame.KEYDOWN: if event.key == pygame.K_RETURN: spiller_ord = input_text.lower() if spiller_ord == "slutt": running = False message = f"Spillet er over! Poeng: {poeng}" elif spiller_ord in brukt_ord: message = "Dette ordet er allerede brukt!" elif spiller_ord[0] != current_word[-1]: message = f"Feil! Må starte med '{current_word[-1]}'" elif spiller_ord not in ordliste: message = "Ordet finnes ikke i ordlisten!" else: poeng += 1 current_word = spiller_ord brukt_ord.append(current_word) message = f"Riktig! Poeng: {poeng}" input_text = "" elif event.key == pygame.K_BACKSPACE: input_text = input_text[:-1] else: input_text += event.unicode # Tegn alt draw_text("Skriv ord og trykk Enter (slutt for å avslutte)", BLACK, 20, 20) draw_text(f"Ditt input: {input_text}", BLACK, 20, 100) draw_text(f"{message}", GREEN if "Riktig" in message else RED, 20, 180) draw_text(f"Poeng: {poeng}", BLACK, 20, 260) pygame.display.flip() clock.tick(30) pygame.quit() sys.exit() if __name__ == "__main__": main()
