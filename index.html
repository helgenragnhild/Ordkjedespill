import random # Liste med noen startord ordliste = ["apple", "elephant", "tiger", "rabbit", "tree", "eagle", "dog", "giraffe"] def spill(): poeng = 0 brukt_ord = [] # Velg et tilfeldig startord forrige_ord = random.choice(ordliste) brukt_ord.append(forrige_ord) print("Velkommen til ordspill!") print(f"Startord: {forrige_ord}") while True: nytt_ord = input(f"Skriv et ord som starter med '{forrige_ord[-1]}': ").lower().strip() if nytt_ord == "slutt": print(f"Spillet er over! Du fikk {poeng} poeng.") break if nytt_ord in brukt_ord: print("Dette ordet er allerede brukt. Prøv et annet ord.") elif nytt_ord[0] != forrige_ord[-1]: print(f"Feil startbokstav! Ordet må starte med '{forrige_ord[-1]}'.") else: poeng += 1 brukt_ord.append(nytt_ord) forrige_ord = nytt_ord print(f"Riktig! Poeng: {poeng}") spill()
